001package org.anarres.parallelgzip;
002
003import java.util.concurrent.ArrayBlockingQueue;
004import java.util.concurrent.ExecutorService;
005import java.util.concurrent.Executors;
006import java.util.concurrent.ThreadFactory;
007import java.util.concurrent.ThreadPoolExecutor;
008import java.util.concurrent.TimeUnit;
009import java.util.concurrent.atomic.AtomicLong;
010import javax.annotation.Nonnull;
011
012/**
013 *
014 * @author shevek
015 */
016public class ParallelGZIPEnvironment {
017
018    private static class ThreadPoolHolder {
019
020        private static final ExecutorService EXECUTOR;
021
022        static {
023            ThreadFactory threadFactory = new ThreadFactory() {
024                private final ThreadFactory defaultThreadFactory = Executors.defaultThreadFactory();
025                private final AtomicLong counter = new AtomicLong(0);
026
027                @Override
028                public Thread newThread(@Nonnull Runnable r) {
029                    Thread thread = defaultThreadFactory.newThread(r);
030                    thread.setName("parallelgzip-" + counter.getAndIncrement());
031                    thread.setDaemon(true);
032                    return thread;
033                }
034            };
035            // int nthreads = Math.max(Runtime.getRuntime().availableProcessors() - 1, 1);
036            int nthreads = Runtime.getRuntime().availableProcessors();
037            ThreadPoolExecutor executor = new ThreadPoolExecutor(nthreads, nthreads,
038                    1L, TimeUnit.SECONDS,
039                    new ArrayBlockingQueue<Runnable>(nthreads * 20),
040                    threadFactory,
041                    new ThreadPoolExecutor.CallerRunsPolicy());
042            executor.allowCoreThreadTimeOut(true);
043            EXECUTOR = executor;
044        }
045    }
046
047    @Nonnull
048    public static ExecutorService getSharedThreadPool() {
049        return ThreadPoolHolder.EXECUTOR;
050    }
051}
